00177 inline double MTRand::randNorm( const double& mean, const double& variance )
00178 {
00179         // Return a real number from a normal (Gaussian) distribution with given
00180         // mean and variance by Box-Muller method
00181         double r = sqrt( -2.0 * log( 1.0-randDblExc()) ) * variance;
00182         double phi = 2.0 * 3.14159265358979323846264338328 * randExc();
00183         return mean + r * cos(phi);

00187 {
00188         // Pull a 32-bit integer from the generator state
00189         // Every other access function simply transforms the numbers extracted here
00190         
00191         if( left == 0 ) reload();
00192         --left;
00193                 
00194         register uint32 s1;
00195         s1 = *pNext++;
00196         s1 ^= (s1 >> 11);

00230 inline void MTRand::seed( uint32 *const bigSeed, const uint32 seedLength )
00231 {
00232         // Seed the generator with an array of uint32's
00233         // There are 2^19937-1 possible initial states.  This function allows
00234         // all of those to be accessed by providing at least 19937 bits (with a
00235         // default seed length of N = 624 uint32's).  Any bits above the lower 32
00236         // in each element are discarded.
00237         // Just call seed() if you want to get array from /dev/urandom
00238         initialize(19650218UL);
00239         register int i = 1;
00240         register uint32 j = 0;
00241         unsigned fixN = N;
00242         register int k = ( fixN > seedLength ? fixN : seedLength );
00243         for( ; k; --k )
00244         {
00245                 state[i] =
00246                         state[i] ^ ( (state[i-1] ^ (state[i-1] >> 30)) * 1664525UL );
00247                 state[i] += ( bigSeed[j] & 0xffffffffUL ) + j;
00248                 state[i] &= 0xffffffffUL;
00249                 ++i;  ++j;
00250                 if( i >= N ) { state[0] = state[N-1];  i = 1; }
00251                 if( j >= seedLength ) j = 0;
00252         }
00253         for( k = N - 1; k; --k )
00254         {
00255                 state[i] =
00256                         state[i] ^ ( (state[i-1] ^ (state[i-1] >> 30)) * 1566083941UL );
00257                 state[i] -= i;
00258                 state[i] &= 0xffffffffUL;
00259                 ++i;
00260                 if( i >= N ) { state[0] = state[N-1];  i = 1; }
00261         }
00262         state[0] = 0x80000000UL;  // MSB is 1, assuring non-zero initial array
00263         reload();
00264 }

00269         // Seed the generator with an array from /dev/urandom if available
00270         // Otherwise use a hash of time() and clock() values
00271         
00272         // First try getting an array from /dev/urandom
00273         FILE* urandom = fopen( "/dev/urandom", "rb" );

00291 inline void MTRand::initialize( const uint32 seed )
00292 {
00293         // Initialize generator state with seed
00294         // See Knuth TAOCP Vol 2, 3rd Ed, p.106 for multiplier.
00295         // In previous versions, most significant bits (MSBs) of the seed affect
00296         // only MSBs of the state array.  Modified 9 Jan 2002 by Makoto Matsumoto.
00297         register uint32 *s = state;
00298         register uint32 *r = state;
00299         register int i = 1;
00300         *s++ = seed & 0xffffffffUL;
00301         for( ; i < N; ++i )
00302         {
00303                 *s++ = ( 1812433253UL * ( *r ^ (*r >> 30) ) + i ) & 0xffffffffUL;
00304                 r++;
00305         }

00309 inline void MTRand::reload()
00310 {
00311         // Generate N new values in state
00312         // Made clearer and faster by Matthew Bellew (matthew.bellew@home.com)
00313         register uint32 *p = state;
00314         register int i;
00315         for( i = N - M; i--; ++p )
00316                 *p = twist( p[M], p[0], p[1] );
00317         for( i = M; --i; ++p )
00318                 *p = twist( p[M-N], p[0], p[1] );
00319         *p = twist( p[M-N], p[0], state[0] );
00320 
00321         left = N, pNext = state;
00322 }

00331         static uint32 differ = 0;  // guarantee time-based seeds will change
